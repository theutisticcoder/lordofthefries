let camera, map; const InitGame = async () => { InitWindow(1600, 900); camera = new Camera3D(); camera.position = new Vector3({x: 5,y: 4,z: 6}) // Camera position camera.target = new Vector3({x: 0,y: 0,z: 0 }); // Camera looking at point camera.up = new Vector3 ({x:0,y: 1, z:0}) ; // Camera up vector (rotation towards target) camera.fovy = 45; // Camera field-of-view Y camera.projection = CAMERA_PERSPECTIVE; map = LoadModel("room.glb"); } const UpdateGame = async () => { UpdateCamera(camera, CAMERA_FIRST_PERSON); BeginDrawing(); ClearBackground(RAYWHITE); BeginMode3D(camera); DrawModel(map, new Vector3({x:5,y:4,z:5}), 5, GRAY); EndMode3D(); EndDrawing(); }
